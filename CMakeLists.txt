cmake_minimum_required(VERSION 3.10)
project(qslim VERSION 2.0)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Options
option(USE_OPENGL "Build with OpenGL support" OFF)
option(USE_XFORMS "Build with XForms support" OFF)
option(USE_LIBTIFF "Build with LibTIFF support" OFF)
option(USE_LIBPNG "Build with LibPNG support" OFF)

# Dependencies
if(USE_OPENGL)
    find_package(OpenGL REQUIRED)
    find_package(X11 REQUIRED)
    set(MIX_HAVE_OPENGL 1)
    set(MIX_HAVE_POLYOFFSET 1)
    set(MIX_HAVE_VERTARRAYS 1)
endif()

if(USE_XFORMS)
    find_path(XFORMS_INCLUDE_DIR forms.h)
    find_library(XFORMS_LIBRARY forms)
    if(XFORMS_INCLUDE_DIR AND XFORMS_LIBRARY)
        message(STATUS "Found XForms: ${XFORMS_LIBRARY}")
        set(MIX_USE_X11_FORMS 1)
    else()
        message(FATAL_ERROR "XForms not found")
    endif()
endif()

# Configuration
set(HAVE_UNISTD_H 1)
set(MIX_ANSI_IOSTREAMS 1)
set(MIX_HAVE_FABSF 1)
set(MIX_HAVE_RINT 1)
set(MIX_HAVE_GETRUSAGE 1)
set(MIX_HAVE_RANDOM 1)

add_definitions(-DHAVE_MIXCFG_H -DMIX_ANSI_IOSTREAMS -DHAVE_UNISTD_H -DMIX_HAVE_FABSF -DMIX_HAVE_RINT -DMIX_HAVE_GETRUSAGE -DMIX_HAVE_RANDOM)

if(USE_OPENGL)
    add_definitions(-DMIX_HAVE_OPENGL -DMIX_HAVE_POLYOFFSET -DMIX_HAVE_VERTARRAYS)
endif()

# Create a dummy mixcfg.h since stdmix.h includes it
file(WRITE ${CMAKE_BINARY_DIR}/mixcfg.h "// Generated by CMake\n")

include_directories(
    ${CMAKE_SOURCE_DIR}/mixkit/src
    ${CMAKE_BINARY_DIR}
)

add_subdirectory(mixkit/src)
add_subdirectory(tools/qslim)
add_subdirectory(tools/filters)
add_subdirectory(examples)

enable_testing()
add_subdirectory(tests)
